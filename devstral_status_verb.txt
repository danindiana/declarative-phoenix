```pythonimport subprocessimport redef cmd_status(manifest):    def get_systemd_status(name):        try:            result = subprocess.run(['systemctl', 'is-active', name], capture_output=True, text=True)            if result.returncode == 0:                return result.stdout.strip()            else:                return "MISSING"        except Exception:            return "UNKNOWN"    def get_docker_status(name):        try:            result = subprocess.run(['docker', 'inspect', name, '--format', '{{.State.Status}}'],                                    capture_output=True, text=True)            if result.returncode == 0 and result.stdout.strip():                return result.stdout.strip()            else:                return "MISSING"        except Exception:            return "UNKNOWN"    def get_ufw_status(port):        try:            result = subprocess.run(['ufw', 'status'], capture_output=True, text=True)            if f"{port}" in result.stdout and re.search(rf"\b{port}\b", result.stdout):                return "OK"            else:                return "MISSING"        except Exception:            return "UNKNOWN"    def colorize(text, color_code):        return f"\033[{color_code}m{text}\033[0m"    services = []    # Process systemd entries    for entry in manifest.get('systemd', []):        name = entry['name']        desired_state = entry['state'] if 'state' in entry else "UNKNOWN"        actual_state = get_systemd_status(name)        status = "OK" if (desired_state == actual_state and actual_state != "MISSING") else (            "DRIFT" if actual_state != "MISSING" else "MISSING")        services.append(("systemd", name, desired_state, actual_state, status))    # Process docker entries    for entry in manifest.get('docker', []):        name = entry['name']        image = entry.get('image', "")        ports = entry.get('ports', {})        port_str = ", ".join([f"{k}->{v}" for k, v in ports.items()])        try:            actual_state = get_docker_status(name)            status = "OK" if actual_state != "MISSING" else "MISSING"        except Exception:            actual_state = "UNKNOWN"            status = "UNKNOWN"        services.append(("docker", name, image + " (" + port_str + ")", actual_state, status))    # Process ufw entries    for entry in manifest.get('ufw', []):        port = entry['port']        protocol = entry.get('protocol', "")        action = entry.get('action', "")        actual_state = get_ufw_status(port)        status = "OK" if actual_state == "OK" else "MISSING"        services.append(("ufw", f"{port}/{protocol} ({action})", "", actual_state, status))    # Print table header    print(f"{'TYPE':<10} {'NAME':<35} {'DESIRED':<20} {'ACTUAL':<15} STATUS")    # Print services with colorized status    ok_count = 0    drift_count = 0    missing_count = 0    for service in services:        type_, name, desired, actual, status = service        if status == "OK":            colored_status = colorize(status, '32')  # Green            ok_count += 1        elif status == "DRIFT" or status == "MISSING":            colored_status = colorize(status, '31')  # Red            if status == "DRIFT":                drift_count += 1            else:                missing_count += 1        else:  # UNKNOWN            colored_status = colorize(status, '33')  # Yellow        print(f"{type_:<10} {name:<35} {desired:<20} {actual:<15} {colored_status}")    # Print summary line    print(f"\n{ok_count} ok, {drift_count} drift, {missing_count} missing")# Example manifest for testingmanifest = {    'systemd': [        {'name': 'nginx-proxy', 'state': 'running'},        {'name': 'backup-timer', 'state': 'enabled'}    ],    'docker': [        {'name': 'open-webui', 'image': 'ghcr.io/open-webui/open-webui:main', 'ports': {'3000': '3000'}}    ],    'ufw': [        {'port': 80, 'protocol': 'tcp', 'action': 'ALLOW'},        {'port': 443, 'protocol': 'tcp', 'action': 'DENY'}    ]}# Run the function with example manifestcmd_status(manifest)```
